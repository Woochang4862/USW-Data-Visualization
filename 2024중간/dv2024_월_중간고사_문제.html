# Y의 2차 기울기 함수 Y2 (즉, Y1의 기울기 함수)를 추가로 plot하시오

# 주의1) 원래 강의자료의 plot_slope 함수에 return이 추가됨

def plot_slope(X1,Y1,ax1):
    Xs=X1[1:]-X1[:-1]
    Ys=Y1[1:]-Y1[:-1]
    _=ax1.plot(X1[1:], Ys/Xs, 'b')
    
    return Ys/Xs
    
    
fig=plt.figure(figsize=(4,3), dpi=100)
ax=fig.subplots()

X=np.linspace(-4,3,300)
Y=(X+3)*(X+1)*(X-2)

_=ax.plot(X,Y, color='r')
Y1=plot_slope(X, Y , ax)
