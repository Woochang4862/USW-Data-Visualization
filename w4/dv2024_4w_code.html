In [4]:
fig=plt.figure(figsize=(12,5), dpi=100)
axs=fig.subplots(1,3)

def bar_width(ax,data,w):
    ax.bar(range(len(data)),data, width=w)
    ax.set_title('width: {}'.format(w))
    
data = np.array([5,25,50,20])
bar_width(axs[0], data, w=0.8) # default width: 0.8
bar_width(axs[1], data, w=0.5)
bar_width(axs[2], data, w=1)
