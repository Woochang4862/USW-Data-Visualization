# 평균 2, 표준편차 2인 정규분포에서 3000개의 값을 뽑아서 히스토그램을 그렸다.
# 해당분포의 pdf 함수를 추가로 그리시오
# h1) plot line color='k'
# h2) pdf 함수의 x range: -4 ~ 8

fig=plt.figure(figsize=(5,4), dpi=100)
ax=fig.subplots()

_=ax.hist(np.random.normal(2,2,size=3000), bins=30, edgecolor='k', density=True)

## 코드작성
def pdf(X, mu, sigma):
    a = 1/(sigma * np.sqrt(2*np.pi))
    b = -1/(2*(sigma**2))
    return a * np.exp(b * ((X - mu)** 2))

X = np.linspace(-4, 8, 1000)
_=ax.plot(X, pdf(X, 2, 2), color = 'k')
## 코드작성 끝
