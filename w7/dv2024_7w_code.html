fig=plt.figure(figsize=(12,5), dpi=100)
ax1, ax2=fig.subplots(1,2)

def pdf(X, mu, sigma):
    a = 1/(sigma * np.sqrt(2*np.pi))
    b = -1/(2*(sigma**2))
    return a * np.exp(b * ((X - mu)** 2))

X=np.linspace(-6,6,1000)

## ax1
Y=pdf(X, 0, 2)
_=ax1.plot(X, Y,'o:r')
Y=pdf(X, 0, 1)
_=ax1.plot(X, Y, 'o--b')

## ax2
Y=pdf(X, 0, 2)
_=ax2.plot(X, Y,'o:r', markevery=25)

_=ax2.plot(X, Y, 'o--b', markevery=25, alpha=0.3)
